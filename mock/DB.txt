tbl_mock_test {
	id integer pk increments
	type varchar
	title varchar
	total_marks integer
	duration text
	languages JSONB // ["English","Gujarati"]
	test_json JSONB // nullable
	start_date timestamp
	end_date timestamp
	zip_url varchar //nullable
    answer_pdf varchar //nullable
}

tbl_mock_test_student {
	id integer pk increments
	student_id integer > childid
	mock_test_id integer > tbl_mock_test.id
	test_start_time timestamp
	result_json JSONB nullable 
	marks integer nullable 
	status - (0 - ongoing , 1 - attempted)
	time_taken text
	created_at timestamp
	updated_at timestamp
}
